37 /**
38  * \ingroup group_gl_flexible_type
39  *
40  * The flexible_type is an automatic, self-managing union between the
41  * following types.
42  * - \ref flex_int
43  * - \ref flex_float
44  * - \ref flex_string
45  * - \ref flex_vec
46  * - \ref flex_list
47  * - \ref flex_dict
48  * - \ref flex_image
49  * - \ref flex_undefined
50  *
51  * It is nearly every operator overloaded which
52  * allows it to behave very similarly to python data types.
53  *
54  * The internal representation of the flexible_type is simply:
55  * - 8 bytes of (flex_int, flex_double, flex_string* and flex_vec*)
56  * - 1 byte tag corresponding to a member of the enumeration \ref flex_type_enum
57  *
58  * \code
59  * flexible_type f = 1;
60  * flexible_type f2 = 2;
61  * f += f2; // f is now 3
62  * f += 1; // f is now 4
63  * if (f == 4) std::cout << "!";
64  * f = float(f); // convert f to float
65  * f += 2.5; // f is now 6.5
66  *
67  * // convert to string
68  * f = std::string(f); f is now "5"
69  * f += "hello"; // f is now "5hello"
70  *
71  * // vector test
72  * f = {1.1, 2.2}; // f is now a vector if {1.1, 2.2}
73  * f.push_back(3.3); // f is now a vector if {1.1, 2.2, 3.3}
74  * f.clear(); // f is now an empty vector
75  *
76  * for (flexible_type i = 0;i < 10; ++i) {
77  * f.push_back(i);
78  * }
79  * // f is now a vector from 1.0 to 10.0
80  *
81  * f = std::string("hello"); // f is now a string again
82  * f.mutable_get<flex_string>() = "boo"; // this gets a reference to the underlying storage
83  * f.mutable_get<flex_int>() = 5; // this will implode at runtime
84  * \endcode
85  *
86  * ### Type Information and Contents
87  * The type information for the contents of the flexible type can be obtained in
88  * three ways.
89  * - \ref flexible_type::get_type() Returns a member of the enumeration flex_type_enum
90  * - \ref flexible_type::which() Returns a number corresponding to size_t(get_type())
91  * Provided for compatibility with boost::variant
92  * - \ref flexible_type::type() Returns a std::type_index object corresponding to
93  * the typeid of the stored object.
94  * Provided for compatibility with boost::variant
95  *
96  * Knowing the type of the contents, the \ref flexible_type::get() function
97  * can be used to extract a reference to the content.
98  *
99  * \code
100  * flexible_type f(5);
101  * f.mutable_get<flex_int>() = 10;
102  * std::cout << f.get<flex_int>();
103  * \endcode
104  *
105  * If the incorrect type is provided, the \ref flexible_type::get() function
106  * will throw a runtime exception.
107  *
108  * A large number of casting operators are also provided and these work as expected.
109  * \code
110  * (int)f // casts doubles and integers to a integer. strings and vectors will fail.
111  * (float)f // casts doubles and integers to a double . strings and vectors will fail.
112  * (string)f // returns a printable string representation of all types.
113  * \endcode
114  *
115  * However, there are certain ambiguous cases whch do cause issues when
116  * using the explicit cast so, the \ref flexible_type::to() function is generally
117  * preferred:
118  * \code
119  * f.to<int>()
120  * f.to<float>()
121  * f.to<string>()
122  * \endcode
123  *
124  * ### Undefined
125  * A special undefined type is provided which is useful for identifying missing
126  * values (or the None value/type in Python). Such a value can be created using:
127  * \code
128  * flexible_type f = flex_undefined();
129  * // or
130  * flexible_type f = FLEX_UNDEFINED;
131  * \endcode
132  *
133  * ### Operators
134  * Practically every operator is overloaded, and the behaviors are relatively
135  * intuitive. See the documentation for each operator to determine legality.
136  * There are also special operator overloads to simplify access to the vector.
137  * For instance, flexible_type::operator[], flexible_type::clear() and
138  * flexible_type::push_back().
139  * In all cases, a runtime exception is thrown if an invalid type combination
140  * is encountered.
141  *
142  * The basic rule of the flexible_type is that the internal type is always
143  * preserved. Only an assignment can change the internal type of the object.
144  * Also, for consistency, operations between two flexible types always return
145  * a type corresponding to the flexible_type on the left of the operation.
146  * i.e.
147  * \code
148  * flexible_type f_int(5);
149  * flexible_type f_float(2.1);
150  *
151  * flexible_type f_res = f_int * f_float;
152  * // the type of f_res is an integer
153  *
154  * f_res = f_float * f_int;
155  * // the type of f_res is a float
156  * \endcode
157  *
158  * This is true also for operations where the 2nd argument is an arbitrary type.
159  *
160  * \code
161  * flexible_type f_float(2.1);
162  *
163  * flexible_type f_res = f_float + 2;
164  * // the type of f_res is a float
165  * \endcode
166  *
167  * The only exception to the rule is when the first argument is not a flexible type
168  * and the 2nd argument is a flexible type. In which case, the return type is the
169  * a flexible type corresponding to the type on the right.
170  *
171  * \code
172  * flexible_type f_float(2.1);
173  *
174  * flexible_type f_res = 2 + f_float;
175  * // the type of f_res is a float
176  * \endcode
177  *
178  * In summary:
179  * Where flexible_type_A is used to denote a flexible_type where the internal
180  * contents are of type A, and OP is an arbitrary operator,
181  * - flexible_type_A OP flexible_type_B ==> flexible_type_A
182  * - flexible_type_A OP T ==> flexible_type_A
183  * - T OP flexible_type_A ==> flexible_type_A
184  *
185  * This does mean that the binary operators are not commutative when performed
186  * across different types. ex:
187  * - (integer * float) produces an integer
188  * - (float * integer) produces an float
189  *
190  *
191  * ### Visitors
192  * Of the most powerful feature of the flexible_type, is the visitor mechanism
193  * which is designed in a very similar manner to that of the boost::variant.
194  * See \ref flexible_type::apply_visitor(Visitor) or
195  * \ref flexible_type::apply_mutating_visitor(Visitor) for details.
196  *
197  *
198  * ### Capability Checking
199  * Sometimes it can be useful to be able to query (at runtime, or at compile
200  * time) whether certain flexible_type operations will succeed or not. A number
201  * of useful functions are provided.
202  *
203  * Runtime Capability Queries:
204  * - \ref flex_type_has_binary_op . Checks if binary operators between certain
205  * types are supported.
206  * - \ref flex_type_is_convertible . Checks if certain type conversions can
207  * be performed.
208  *
209  * Compile-time Capability Queries:
210  * - \ref is_valid_flex_type . Checks at compile time if a type is one of the
211  * internal flexible_type types.
212  * - \ref enum_to_type . Gets the internal type from an enumeration value.
213  * - \ref type_to_enum . Gets the enumeration value from the internal type.
214  * - \ref has_direct_conversion_to_flexible_type . Can be casted to a flexible_type.
215  *
216  * ### Performance
217  * Performance of the flexible_type is generally extremely good. With modern
218  * compilers (sometimes with inlining limit bumped up) can sometimes produce
219  * code which is equivalent to, or runs as fast as the version using native
220  * types especially if certain type constraints are set-up before hand.
221  * For instance:
222  *
223  * \code
224  * if (x.get_type() == flex_type_enum::INTEGER) {
225  * ... do a whole bunch of stuff on x ...
226  * }
227  * \endcode
228  *
229  * Integers and Floating point values are stored in-place inside the
230  * flexible_type. All other types are somewhat more complex (strings,
231  * vectors etc) and are optimized by copy on write. i.e.
232  * \code
233  * flexible_type a = "hello world";
234  * flexible_type b = a;
235  * \endcode
236  * Both b and a will reference the same "hello world" until either one of them
237  * gets mutated. As such, all inplace operations
238  * (\ref get(), \ref apply_visitor(Visitor)) require const, but also have a
239  * mutating version (\ref mutating_get(), \ref apply_mutating_visitor(Visitor)).
240  */

530  /**
531  * Executes an apply visitor on the flexible type.
532  * The apply visitor object must have a function void operator()(T& t) for
533  * every type T the flexible type can contain. The operator() function
534  * can be itself templated for convenience. For instance, the following
535  * visitor will add the value "1" to the flexible_type if the flexible_type
536  * contains an integer, and will ignore otherwise.
537  * \code
538  * struct add_one_to_integer {
539  * // do nothing
540  * template <typename T>
541  * void operator()(T& t) { }
542  *
543  * void operator()(flex_int& t) { t += 1; }
544  * };
545  * \endcode
546  *
547  * The visitor can return values but all versions of the operator() function
548  * must return exactly the same type. For instance:
549  *
550  * \code
551  * struct add_one_and_return {
552  * // do nothing
553  * template <typename T>
554  * int operator()(T& t) { return 0; }
555  *
556  * int operator()(flex_int& t) { t += 1; return t; }
557  * };
558  * \endcode
559  *
560  * In which case, apply_mutating_visitor will return a value:
561  * \code
562  * int ret = flex.apply_mutating_visitor(add_one_and_return());
563  * \endcode
564  *
565  * This function assumes that the function will mutate the value. If
566  * the function does not mutate the value, the alternative
567  * \ref apply_visitor(Visitor) should be used.
568  */
569  template <typename Visitor>
570  auto apply_mutating_visitor(Visitor visitor) -> decltype(visitor(prototype_flex_int));
