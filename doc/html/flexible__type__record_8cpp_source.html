53 field_id_type flexible_type_record::add_field(field_id_type field_id, flexible_type data) {
54  // confirm that this field_id doesn't already exist in the vector
55  std::pair<size_t, flexible_type*> ptr = get_ptr();
56  for (size_t i = 0;i < ptr.first; ++i) {
57  if (ptr.second[i].tag() == field_id) {
58  // update existing value
59  ptr.second[i] = data;
60  // refresh the tag value
61  ptr.second[i].tag() = field_id;
62  return (field_id_type)(-1);
63  }
64  }
65  // no it does not exist. create a new element
66  flexible_type* newel = add_one_element();
67  (*newel) = data;
68  newel->tag() = field_id;
69  return field_id;
70 }

72 void flexible_type_record::remove_field(field_id_type field_id) {
73  // first find the element
74  bool found = false;
75  size_t foundat = 0;
76  std::pair<size_t, flexible_type*> ptr = get_ptr();
77  for (size_t i = 0;i < ptr.first; ++i) {
78  if (ptr.second[i].tag() == field_id) {
79  foundat = i;
80  found = true;
81  break;
82  }
83  }
84  if (!found) return;
85 
86  if (ptr.first == 1) {
87  // only 1 element, clear the array than
88  clear();
89  } else if (ptr.first == 2) {
90  // we need to downgrade to 1 element
91  // read out the other element
92  flexible_type otherval = std::move(ptr.second[!foundat]);
93  // preserve the tag
94  otherval.tag() = ptr.second[!foundat].tag();
95  resize_and_clear(1);
96  //move into the new array
97  ptr = get_ptr();
98  // remember to reserve the tag.
99  ptr.second[0] = std::move(otherval);
100  ptr.second[0].tag() = otherval.tag();
101  } else {
102  // we have a full array.
103  // move the last element to it so we can just delete the last element
104  if (ptr.first - 1 != foundat) {
105  // if we are not the last element in the array
106  // remember to preserve the tag
107  ptr.second[foundat] = std::move(ptr.second[ptr.first - 1]);
108  ptr.second[foundat].tag() = ptr.second[ptr.first - 1].tag();
109  }
110  // delete the last element
111  ptr.second[ptr.first - 1].~flexible_type();
112  // decrease the size of the array
113  values.ptr.first--;
114  values.ptr.second = (flexible_type*)realloc(values.ptr.second,
115  sizeof(flexible_type) * values.ptr.first);
116 
117  }
118 }

152 flexible_type* flexible_type_record::add_one_element() {
153  // inserts an element
154  if (is_empty()) {
155  // upgrade to single from empty
156  new (&values.single) flexible_type();
157  // set the bits to show I am now single
158  set_single();
159  return &values.single;
160  } else if (is_single()) {
161  // upgrade to array from single
162  // allocate the new array
163  flexible_type* newarr = (flexible_type*)malloc(sizeof(flexible_type) * 2);
164  std::uninitialized_fill_n(newarr, 2, flexible_type());
165  //move the current contents to newarr[0]
166  newarr[0] = std::move(values.single);
167  newarr[0].tag() = values.single.tag();
168  // delete the current value
169  values.single.~flexible_type();
170  // fill the union
171  values.ptr.first = 2;
172  values.ptr.second = newarr;
173  return values.ptr.second + 1;
174  } else if (is_array()) {
175  // resize the array
176  ++values.ptr.first;
177  values.ptr.second = (flexible_type*)realloc(values.ptr.second,
178  sizeof(flexible_type) * values.ptr.first);
179  // initialize the newly created element using placement new
180  new (values.ptr.second + (values.ptr.first - 1)) flexible_type();
181  return values.ptr.second + (values.ptr.first - 1);
182  }
183  return NULL;
184 }
