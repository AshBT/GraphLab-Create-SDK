93 /**
94  * Compute weighted pagerank of the given graph.
95  *
96  * Return a new graph with "pagerank" field on each vertex.
97  *
98  * \param g The input graph
99  * \param iters Number of iterations
100  * \param weighted_field The edge field which contains the weight value.
101  */
102 gl_sgraph weighted_pagerank(const gl_sgraph& g, size_t iters, const std::string& weight_field) {
103  const double RESET_PROB=0.15;
104 
105  // Get rid of unused fields.
106  gl_sgraph g_min = g.select_fields({weight_field});
107 
108  // Count the total out going weight of each vertex into an SFrame.
109  gl_sframe outgoing_weight = g_min.get_edges().groupby({"__src_id"}, {{"total_weight", aggregate::SUM(weight_field)}});
110 
111  // Add the total_weight to the graph as vertex atrribute.
112  // We can update the vertex data by adding the same vertex.
113  gl_sgraph g2 = g_min.add_vertices(outgoing_weight, "__src_id");
114 
115  // Lambda function for triple_apply
116  auto pr_update = [&weight_field](edge_triple& triple)->void {
117  double weight = triple.edge[weight_field];
118  triple.target["pagerank"] += triple.source["pagerank_prev"] * weight / triple.source["total_weight"];
119  };
120 
121  // Initialize pagerank value
122  g2.vertices()["pagerank_prev"] = 1.0;
123 
124  // Iteratively run triple_apply with the update function
125  for (size_t i = 0; i < iters; ++i) {
126 
127  g2.vertices()["pagerank"] = 0.0;
128 
129  logprogress_stream << "Iteration " << (i+1) << std::endl;
130  g2 = g2.triple_apply(pr_update, {"pagerank"});
131 
132  g2.vertices()["pagerank"] = RESET_PROB + (1-RESET_PROB) * g2.vertices()["pagerank"];
133  g2.vertices()["pagerank_prev"] = g2.vertices()["pagerank"];
134  }
135 
136  g2.vertices().remove_column("pagerank_prev");
137  g2.vertices().remove_column("total_weight");
138  return g2;
139 }

142 /**
143  * Return a new graph with no duplicate edges.
144  *
145  * \param g The input graph
146  */
147 gl_sgraph edge_deduplication(const gl_sgraph& g) {
148  std::vector<std::string> edge_fields = g.get_edge_fields();
149 
150  std::map<std::string, aggregate::groupby_descriptor_type> groupby_operators;
151  for (const auto& f: edge_fields) {
152  if (f != "__src_id" && f != "__dst_id") {
153  groupby_operators[f] = aggregate::SELECT_ONE(f);
154  }
155  }
156  gl_sframe edge_dedup = g.get_edges().groupby( {"__src_id", "__dst_id"}, groupby_operators );
157  return gl_sgraph(g.get_vertices(), edge_dedup);
158 }
