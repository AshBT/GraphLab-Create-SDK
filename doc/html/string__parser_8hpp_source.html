50  * \internal
51  * This class defines a string parser which allows the parser writer to define
52  * a list of characters which are not permitted in unquoted strings. Quoted
53  * strings have no restrictions on what characters they can contain.
54  * Usage:
55  * \code
56  * parser_impl::parser_config config;
57  * config.[set stuff up]
58  * rule = parser_impl::restricted_string(config);
59  * \endcode
60  */

74  enum class tokenizer_state {
75  START_FIELD, ESCAPED_CHAR, IN_FIELD, IN_QUOTED_FIELD,
76  ESCAPE_IN_QUOTED_FIELD
77  };
78 // insert a character into the field buffer. resizing it if necessary
79 
80  // This function is called during the actual parsing process
81  template <typename Iterator, typename Context, typename Skipper, typename Attribute>
82  bool parse(Iterator& first, Iterator const& last,
83  Context&, Skipper const& skipper, Attribute& attr) const {
84  boost::spirit::qi::skip_over(first, last, skipper);
85  Iterator cur = first;
86  std::string ret;
87 
88  tokenizer_state state = tokenizer_state::START_FIELD;
89  bool keep_parsing = true;
90  char quote_char = 0;
91  while(keep_parsing && cur != last) {
92  // Next character in file
93  char c = *cur;
94  if(state != tokenizer_state::IN_QUOTED_FIELD &&
95  state != tokenizer_state::ESCAPE_IN_QUOTED_FIELD &&
96  config.restrictions.find(c) != std::string::npos) break;
97  ++cur;
98  switch(state) {
99  case tokenizer_state::START_FIELD:
100  if (c == '\'' || c == '\"') {
101  quote_char = c;
102  state = tokenizer_state::IN_QUOTED_FIELD;
103  } else if (c == config.escape_char) {
104  /* possible escaped character */
105  state = tokenizer_state::ESCAPED_CHAR;
106  } else {
107  /* begin new unquoted field */
108  ret = ret + c;
109  state = tokenizer_state::IN_FIELD;
110  }
111  break;
112 
113  case tokenizer_state::ESCAPED_CHAR:
114  ret = ret + graphlab::unescape_char(c);
115  state = tokenizer_state::IN_FIELD;
116  break;
117 
118  case tokenizer_state::IN_FIELD:
119  /* in unquoted field */
120  if (c == config.escape_char) {
121  /* possible escaped character */
122  state = tokenizer_state::ESCAPED_CHAR;
123  } else {
124  /* normal character - save in field */
125  ret = ret + c;
126  }
127  break;
128 
129  case tokenizer_state::IN_QUOTED_FIELD:
130  /* in quoted field */
131  if (c == config.escape_char) {
132  /* Possible escape character */
133  state = tokenizer_state::ESCAPE_IN_QUOTED_FIELD;
134  }
135  else if (c == quote_char) {
136  if (c == '\"' && config.double_quote) {
137  /* doublequote; " represented by "" */
138  // look ahead one character
139  if (cur + 1 < last && *cur == quote_char) {
140  ret = ret + c;
141  ++cur;
142  break;
143  }
144  }
145  // we are done.
146  keep_parsing = false;
147  }
148  else {
149  /* normal character - save in field */
150  ret = ret + c;
151  }
152  break;
153 
154  case tokenizer_state::ESCAPE_IN_QUOTED_FIELD:
155  ret = ret + graphlab::unescape_char(c);
156  state = tokenizer_state::IN_QUOTED_FIELD;
157  break;
158  }
159  }
160  if (cur == first) return false;
161  else {
162  if (!quote_char) boost::algorithm::trim_right(ret);
163  attr = ret;
164  first = cur;
165  }
166  return true;
167  }
