11 /**
12  * \ingroup unity
13  * The flexible_type_record represents a collection of (field-id, flexible_type)
14  * pairs.
15  *
16  * The internal represention of the flexible type record varies depending on
17  * the number of fields it contains. When it contains just a single field,
18  * it is a single 16-byte entry which represents a flexible_type with the field
19  * information held in the tag in the flexible_type. When the number of
20  * fields is larger than one, it maintained a pair of (size_t, flexible_type*)
21  * which is also 16 bytes, and represents a length, and a pointer to an
22  * array of flexible_type objects. Once again the field id is
23  * maintained in the tag in the flexible_type.
24  */

141  /**
142  * Returns the value of a field if the field exists.
143  * Otherwise, creates a new field with an empty value.
144  *
145  * \note When it creates a new field, the flexible_type is
146  * unaware of the actual type, and is thus undefined. Some care needed
147  * if you are uncertain whether the field exists in the record.
148  */
149  inline flexible_type& operator[](field_id_type s) {
150  std::pair<size_t, flexible_type*> ptr = get_ptr();
151  for (size_t i = 0;i < ptr.first; ++i) {
152  if (ptr.second[i].tag() == s) return ptr.second[i];
153  }
154  // no existing element. create a new one

156  newel->tag() = s;

188  /**
189  * Returns true if this flexible_type_record contains all the fields in
190  * "other" and matches all values; returns false otherwise. Like is_superset_of
191  * but allows UNDEFINED matches everything. i.e. if either UNDEFINED is
192  * defined to be equal to everything.
193  */
194  inline bool is_superset_of_undefined_wildcard(const flexible_type_record& other) const {
195  for (const auto& f : other.fields()) {
196  if (!has_field(f.tag()) ||
197  (f.get_type() != flex_type_enum::UNDEFINED &&
198  (*this)[f.tag()].get_type() != flex_type_enum::UNDEFINED &&
199  (*this)[f.tag()] != f)) return false;
200  }
201  return true;
202  }

216  private:
217  /*
218  * we use the lower two bits of the union to identify if there are no elements,
219  * one element, or more than one element in the record. Specifically,
220  * if the lower two bits are
221  * 00 --> the contents of the records is an array of elements
222  * 01 --> There are no elements in the record
223  * 11 --> There is one element in the record
224  *
225  * This is safe because the the lower 3 bytes of the flexible_type is not
226  * used, and a malloc of the flexible type will necessarily require 8 byte
227  * alignment thus forcing the lower 3 bits to be 0.
228  */
229  union union_type {
230  union_type() : words{0,0} { }
231  ~union_type() { }
232  flexible_type single; // stores a single element
233  std::pair<size_t, flexible_type*> ptr; //stores an array of elements
234  std::pair<size_t, size_t> words; // used to get direct access to the record memory
235  } values;
236
