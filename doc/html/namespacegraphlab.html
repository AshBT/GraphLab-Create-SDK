template<typename T >
using graphlab::is_flexible_type_member = typedef boost::mpl::contains<flexible_type_types, T>

is_flexible_type_member<T>::value is true if T is a type contained by the flexible_type, and is false otherwise.

is_flexible_type_member<size_t>::value // false
is_flexible_type_member<flex_int>::value // true (flex_int == int64_t)
is_flexible_type_member<std::vector<flexible_type>>::value // true
typedef boost::make_recursive_variant< flexible_type, std::shared_ptr<unity_sgraph_base>, dataframe_t, std::shared_ptr<model_base>, std::shared_ptr<unity_sframe_base>, std::shared_ptr<unity_sarray_base>, std::map<std::string, boost::recursive_variant_>, std::vector<boost::recursive_variant_>, boost::recursive_wrapper<function_closure_info> >::type graphlab::variant_type

A variant object that can be communicated between Python and C++ which contains either a

  • flexible_type
  • graph
  • dataframe_t
  • model
  • sframe
  • std::map<variant>
  • std::vector<variant>

See the boost variant documentation for details.

The variant should not be accessed directly. See to_variant and variant_get_value for powerful ways to extract or store values from a variant.
