14 /**
15  * The archive index.
16  *
17  * The archive index file simply comprises of the following:
18  * \code
19  * [archive]
20  * version = 1
21  * num_prefixes = 4
22  * [prefixes]
23  * 0000 = "dir_archive.ini"
24  * 0001 = "objects.bin"
25  * 0002 = "0001"
26  * 0003 = "0002"
27  * \endcode
28  * The prefix section basically lists all the prefixes stored inside the
29  * directory archive. All files in the directory which have their file name
30  * beginning with a prefix is a file belonging to the archive.
31  *
32  * The objects.bin, and dir_archive.ini file is always in the prefix
33  *
34  * Once read into the archive_index_information struct however, the prefixes
35  * will all be absolute paths.
36  */
37 struct archive_index_information {
38  size_t version = (size_t)(-1);
39  std::vector<std::string> prefixes;
40  std::map<std::string, std::string> metadata;
41 };

45 /**
46  * The dir_archive object manages a directory archive. It is an internal
47  * class which provides two basic containers:
48  * - A single file stream object (a general_ifstream / general_ofstream)
49  * which points to an "objects.bin" file in the directory.
50  * - The ability to obtain prefixes (for instance [directory]/0000) which
51  * consumers can then use for other file storage purposes. (for instance,
52  * an sframe could create 0000.sidx, 0000.0001, 0000.0002, etc),
53  *
54  * The directory archive provide management for the prefixes and the objects
55  * as well as directory archive creation / deletion.
56  *
57  * To use:
58  * \code
59  * dir_archive archive;
60  * archive.open_directory_for_write(dir)
61  * oarchive oarc(archive)
62  * oarc << ...
63  * oarc.get_prefix()
64  * etc.
65  * \endcode
66  * Similarly, to read:
67  * \code
68  * dir_archive archive;
69  * archive.open_directory_for_read(dir)
70  * iarchive iarc(archive)
71  * iarc >> ...
72  * iarc.get_prefix()
73  * etc.
74  * \endcode
75  */
