57  field_id_type flexible_type_registry::register_field(std::string name,
58  flex_type_enum type) {
59  // variable name already exists
60  if (_registry_name.count(name)) {
61  logstream(LOG_INFO) << "Failed to registering a field " << name
62  << " of type " << (int)type
63  << " because field already exists" << std::endl;
64  return (field_id_type)(-1);
65  }
66  field_id_type field_id = this->_registered_field_counter;
67  this->_registered_field_counter++;
68  logstream(LOG_INFO) << "Registering a field " << name << " of type "
69  << (int)type << std::endl;
70  _registry_name[name] = field_id;
71  _registry_index_to_name[field_id] = name;
72  _registry_index[field_id] = type;
73  return field_id;
74  }

76  void flexible_type_registry::unregister_field(std::string name) {
77  auto found_name = _registry_name.find(name);
78  if (found_name == _registry_name.end()) {
79  // if name not found, nothing else to do here
80  return;
81  }
82 
83  field_id_type field_id = found_name->second;
84  _registry_name.erase(found_name);
85 
86  // delete from _registry_index_to_name if found
87  auto found_index_to_name_iter = _registry_index_to_name.find(field_id);
88  if (found_index_to_name_iter != _registry_index_to_name.end()) {
89  _registry_index_to_name.erase(found_index_to_name_iter );
90  }
91 
92  // delete from _registry_index if found
93  auto found_index_iter = _registry_index.find(field_id);
94  if (found_index_iter != _registry_index.end()) {
95  _registry_index.erase(found_index_iter);
96  }
97  }
