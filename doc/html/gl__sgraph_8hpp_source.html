19 /**
20  * \ingroup group_glsdk
21  * A scalable graph data structure backed by persistent storage (\ref gl_sframe).
22  *
23  * The SGraph (\ref gl_sgraph) data structure allows arbitrary
24  * dictionary attributes on vertices and edges, provides flexible vertex and
25  * edge query functions, and seamless transformation to and from SFrame.
26  *
27  * ### Construction
28  *
29  * There are several ways to create an SGraph. The simplest way is to make an
30  * empty graph then add vertices and edges with the \ref add_vertices
31  * and \ref add_edges methods.
32  *
33  * \code
34  * gl_sframe vertices { {"vid", {1,2,3} };
35  * gl_sframe edges { {"src", {1, 3}}, {"dst", {2, 2}} };
36  * gl_sgraph g = gl_sgraph().add_vertices(vertices, "vid").add_edges(edges, "src", "dst");
37  * \endcode
38  *
39  * Columns in the \ref gl_sframes that are not used as id fields are assumed
40  * to be vertex or edge attributes.
41  *
42  * \ref gl_sgraph objects can also be created from vertex and edge lists stored
43  * in \ref gl_sframe.
44  *
45  * \code
46  * gl_sframe vertices { {"vid", {1,2,3}, {"vdata" : {"foo", "bar", "foobar"}} };
47  * gl_sframe edges { {"src", {1, 3}}, {"dst", {2, 2}}, {"edata": {0., 0.}} };
48  * gl_sgraph g = gl_sgraph(vertices, edges, "vid", "src", "dst");
49  * \endcode
50  *
51  * ### Usage
52  *
53  * The most convenient way to access vertex and edge data is through
54  * the \ref vertices and \ref edges.
55  * Both functions return a GFrame (\ref gl_gframe) object. GFrame is
56  * like SFrame but is bound to the host SGraph,
57  * so that modification to GFrame is applied to SGraph, and vice versa.
58  *
59  * For instance, the following code shows how to add/remove columns
60  * to/from the vertex data. The change is applied to SGraph.
61  * \code
62  * // add a new edge attribute with const value.
63  * g.edges().add_column("likes", 0);
64  *
65  * // remove a vertex attribute.
66  * g.vertices().remove_column("age");
67  *
68  * // transforms one attribute to the other
69  * g.vertices()["likes_fish"] = g.vertices()["__id"] == "cat";
70  * \endcode
71  *
72  * You can also query for specific vertices and edges using the
73  * \ref get_vertices and \ref get_edges functionality.
74  *
75  * For instance,
76  * \code
77  * gl_sframe selected_edges = g.get_edges(
78  * { {0, UNDEFINED}, {UNDEFINED, 1}, {2, 3} },
79  * { {"likes_fish", 1} });
80  * \endcode
81  * selects out going edges of 0, incoming edges of 1, edge 2->3, such that
82  * the edge attribute "like_fish" evaluates to 1.
83  *
84  * In addition, you can perform other non-mutating \ref gl_sframe operations like
85  * groupby, join, logical_filter in the same way, and the returned object will
86  * be \ref gl_sframe.
87  *
88  * In the case where you want to perform vertex-specified operations,
89  * such as "gather"/"scatter" over the neighborhood of each vertex,
90  * we provide \ref triple_apply which is essentially a
91  * "parallel for" over (Vertex, Edge, Vertex) triplets.
92  *
93  * For instance, the following code shows how to implement the update function
94  * for synchronous pagerank.
95  *
96  * \code
97  *
98  * const double RESET_PROB = 0.15;
99  *
100  * void pr_update(edge_triple& triple) {
101  * triple.target["pagerank"] += triple.source["pagerank_prev"] / triple.source["out_degree"];
102  * }
103  *
104  * gl_sgraph pagerank(const gl_sgraph& g, size_t iters) {
105  *
106  * // Count the out degree of each vertex into an gl_sframe.
107  * gl_sframe out_degree = g.get_edges().groupby("__src_id", {{"out_degree", aggregate::COUNT()}});
108  *
109  * // Add the computed "out_degree" to the graph as vertex attribute.
110  * // We exploit that adding the same vertex will overwrite the vertex data.
111  * gl_sgraph g2 = g.add_vertices(out_degree, "__src_id");
112  *
113  * // Initialize pagerank value
114  * g2.vertices()["pagerank"] = 0.0;
115  * g2.vertices()["pagerank_prev"] = 1.0;
116  *
117  * for (size_t i = 0; i < iters; ++i) {
118  * g2.vertices()["pagerank"] = 0.0;
119  * g2 = g2.triple_apply(pr_update, {"pagerank"});
120  * g2.vertices()["pagerank"] = RESET_PROB + (1 - RESET_PROB) * g2.vertices()["pagerank"];
121  * g2.vertices()["pagerank_prev"] = g2.vertices()["pagerank"];
122  * }
123  *
124  * g2.vertices().remove_column("pagerank_prev");
125  * g2.vertices().remove_column("out_degree");
126  * return g2;
127  * }
128  *
129  * \endcode
130  *
131  * ### Mutability
132  *
133  * \ref gl_sgraph is structurally immutable but data (or field) Mutable.
134  * You can add new vertices and edges, but the operation returns a new \ref
135  Ï
136  * \endcode
137  *
138  * ### Example
139  *
140  * Please checkout graphlab/sdk_example/sgraph_example.cpp for a concrete example.
141  */

240  /**
241  * Return a collection of edges and their attributes.
242  *
243  * This function is used to find edges by vertex IDs,
244  * filter on edge attributes, or list in-out * neighbors of vertex sets.
245  *
246  * \param ids Optional. Array of pairs of source and target vertices, each
247  * corresponding to an edge to fetch. Only edges in this list are returned.
248  * FLEX_UNDEFINED can be used to designate a wild card. For instance, {{1,3},
249  * {2,FLEX_UNDEFINED}, {FLEX_UNDEFINED, 5}} will fetch the edge 1->3, all
250  * outgoing edges of 2 and all incoming edges of 5. ids may be left empty,
251  * which implies an array of all wild cards.
252  *
253  * \param fields Optional. Dictionary specifying equality constraints on
254  * field values. For example, { {"relationship", "following"} }, returns only
255  * edges whose 'relationship' field equals 'following'. FLEX_UNDEFINED can be
256  * used as a value to designate a wild card. e.g. { {"relationship",
257  * FLEX_UNDEFINED} } will find all edges with the field 'relationship'
258  * regardless of the value.
259  *
260  * Example:
261  * \code
262  * gl_sframe edges{ {"__src_id", {0, 0, 1}},
263  * {"__dst_id", {1, 2, 2}},
264  * {"rating", {5, 2, FLEX_UNDEFINED}} };
265  * gl_sgraph g = gl_sgraph().add_edges(edges);
266  *
267  * std::cout << g.get_edges() << std::endl;
268  * std::cout << g.get_edges({}, { {"rating": 5} }) << std::endl;
269  * std::cout << g.get_edges({ {0, 1}, {1, 2} }) << std::endl;
270  * \endcode
271  *
272  * Produces output:
273  * \code{.txt}
274  * Return all edges in the graph.
275  *
276  * +----------+----------+--------+
277  * | __src_id | __dst_id | rating |
278  * +----------+----------+--------+
279  * | 0 | 2 | 2 |
280  * | 0 | 1 | 5 |
281  * | 1 | 2 | None |
282  * +----------+----------+--------+
283  *
284  * Return edges with the attribute "rating" of 5.
285  *
286  * +----------+----------+--------+
287  * | __src_id | __dst_id | rating |
288  * +----------+----------+--------+
289  * | 0 | 1 | 5 |
290  * +----------+----------+--------+
291  *
292  * Return edges 0 --> 1 and 1 --> 2 (if present in the graph).
293  *
294  * +----------+----------+--------+
295  * | __src_id | __dst_id | rating |
296  * +----------+----------+--------+
297  * | 0 | 1 | 5 |

867  /* Triple Apply */
868  /* */
869  /**************************************************************************/
870  /**
871  * Apply a user defined lambda function on each of the edge triples, and returns
872  * the new graph.
873  *
874  * An \ref edge_triple is a simple struct containing source, edge and target
875  * of type std::map<std::string, flexible_type>. The lambda function
876  * is applied once on each of the edge_triple in parallel, with locking on
877  * both source and target vertices to prevent race conditions. The following
878  * pseudo code describes the effect of the function:
879  *
880  * \code
881  * INPUT: G
882  * OUTPUT: G'
883  * G' = copy(G)
884  * PARALLEL FOR (source, edge, target) in G':
885  * LOCK (source, target)
886  * edge_triple triple(source, edge, target)
887  * lambda(triple)
888  * FOR f in mutated_fields:
889  * source[f] = triple.source[f] // if f in source
890  * target[f] = triple.target[f] // if f in target
891  * edge[f] = triple.edge[f] // if f in edge
892  * END FOR
893  * UNLOCK (source, target)
894  * END PARALLEL FOR
895  * RETURN G'
896  * \endcode
897  *
898  * This function enables super easy implementations of common graph
899  * computations like degree_count, weighted_pagerank, connected_component, etc.
900  *
901  * Example
902  *
903  * \code
904  * gl_sframe edges { {"source": {0,1,2,3,4}},
905  * {"dest": {1,2,3,4,0}} };
906  * g = graphlab.SGraph().add_edges(edges, "source", "dest");
907  * g.vertices()['degree'] = 0
908  * std::cout << g.vertices() << std::endl;
909  *
910  * auto degree_count_fn = [](edge_triple& triple)->void {
911  * triple.source["degree"] += 1;
912  * triple.target["degree"] += 1;
913  * };
914  *
915  * g2 = g.triple_apply(degree_count_fn, {"degree"});
916  * std::cout << g2.vertices() << std::endl;
917  * \endcode
918  *
919  * Produces output:
920  *
921  * \code{.txt}
922  *
923  * Vertices of g
924  * +------+--------+
925  * | __id | degree |
926  * +------+--------+
927  * | 0 | 0 |
928  * | 2 | 0 |
929  * | 3 | 0 |
930  * | 1 | 0 |
931  * | 4 | 0 |
932  * +------+--------+
933  *
934  * Vertices of g2
935  * +------+--------+
936  * | __id | degree |
937  * +------+--------+
938  * | 0 | 2 |
939  * | 2 | 2 |
940  * | 3 | 2 |
941  * | 1 | 2 |
942  * | 4 | 2 |
943  * +------+--------+
944  *
945  * \endcode
946  *
947  * \note mutated fields must be pre-allocated before triple_apply.
948  *
949  * \see edge_triple
950  * \see lambda_triple_apply_fn
951  */
